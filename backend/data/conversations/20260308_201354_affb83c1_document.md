import re

FILLER_PATTERNS = [
    r"^sure[,!\s]*",
    r"^absolutely[,!\s]*",
    r"^here's the answer[,!\s]*",
    r"^let me explain[,!\s]*"
]

def clean_text(text: str) -> str:
    text = text.strip()

    # Remove filler starts
    for pattern in FILLER_PATTERNS:
        text = re.sub(pattern, "", text, flags=re.IGNORECASE)

    # Normalize whitespace
    text = re.sub(r"\n{3,}", "\n\n", text)
    text = re.sub(r"[ \t]{2,}", " ", text)

    return text.strip()

def detect_code(text: str) -> bool:
    return "```" in text or re.search(r"\b(def|class|function|import)\b", text)

from utils.tokenizer import count_tokens

def preprocess_messages(messages: list[Message]) -> list[CleanMessage]:
    processed = []

    for msg in messages:
        cleaned = clean_text(msg.content)
        tokens = count_tokens(cleaned)
        is_code = detect_code(cleaned)

        processed.append(
            CleanMessage(
                role=msg.role,
                content=cleaned,
                timestamp=msg.timestamp,
                token_count=tokens,
                is_code=is_code
            )
        )

    return processed

  const messages = [];

  let index = 0;

  nodes.forEach((node) => {

    if (!text || text.length < 10) return;

function extractCodeBlocks(node) {
  const blocks = [];
  node.querySelectorAll("pre, code").forEach((el) => {
    const text = el.innerText.trim();
    if (text) blocks.push(text);
  });
  return blocks;
}

function extractTextBlocks(node) {
  const blocks = [];
  node.childNodes.forEach((child) => {
    if (child.nodeType === Node.TEXT_NODE) {
      const text = child.textContent.trim();
      if (text) blocks.push(text);
    }
  });
  return blocks;
}

function scrapeConversation() {
  const messages = [];

  // ChatGPT structure currently uses article
  const nodes = document.querySelectorAll("article");

  nodes.forEach((node, i) => {
    const rawText = node.innerText.trim();
    if (!rawText) return;

    const codeBlocks = extractCodeBlocks(node);
    const textBlocks = extractTextBlocks(node);

    messages.push({
      index: i,

      // raw metadata (no inference here)
      role:
        node.getAttribute("data-message-author-role") ||
        "unknown",

      raw_text: rawText,

      text_blocks: textBlocks,
      code_blocks: codeBlocks,

      has_code: codeBlocks.length > 0,
      char_length: rawText.length,

      // optional but useful for debugging / future
      html_snapshot: node.innerHTML.slice(0, 2000)
    });
  });

  return messages;
}

  const blocks = [];
